       result = read_all_state(ctrl)
       if result is None:

       # Update positions (existing property setters)
       self.pos_a = f"{a:.1f}"
       self.pos_b = f"{b:.1f}"
       self.pos_c = f"{c:.1f}"
       self.pos_d = f"{d:.1f}"
       # Update plot data if applicable (keep existing plot append logic)
